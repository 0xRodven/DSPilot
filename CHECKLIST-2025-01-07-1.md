### Fonction blurName
```typescript
function blurName(name: string): string {
  return name
    .split(" ")
    .map((part) => (part.length > 0 ? `${part[0]}***` : ""))
    .join(" ")
}
// "Jean Dupont" → "J*** D***"
```
